# Sidebar Tree View Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a collapsible left sidebar showing the full Task → Developer → Agent → Sub-Agent hierarchy as a nested tree, independent of canvas filters, with click-to-select-and-pan and full localStorage persistence.

**Architecture:** Single file (`index.html`) — add CSS, HTML, and JS in the three existing sections. Sidebar has its own `renderSidebar()` lifecycle, completely decoupled from the filter-driven `render()` function. All state (visibility + expand) lives in localStorage.

**Tech Stack:** Vanilla HTML/CSS/JS, no dependencies, no build step.

---

## Files

- Modify: `index.html` — all changes are in this one file
  - `<style>` section: add sidebar, toggle, row, active CSS
  - `<body>` section: wrap `#canvas` in `#app-body`, add `#tree-sidebar`
  - `<script>` section: add sidebar state, `renderSidebar()`, `panToNode()`, `updateSidebarActive()`

```js
// ── Sidebar visibility ────────────────────────────────────────────────────
let sidebarVisible = localStorage.getItem('aura-sidebar-visible') !== 'false';

function applySidebarVisibility() {
  const sidebar = document.getElementById('tree-sidebar');
  const toggle  = document.getElementById('tree-toggle');
  if (sidebarVisible) {
    sidebar.classList.remove('sidebar-hidden');
    toggle.textContent = '‹';
  } else {
    sidebar.classList.add('sidebar-hidden');
    toggle.textContent = '›';
  }
}

function toggleSidebar() {
  sidebarVisible = !sidebarVisible;
  localStorage.setItem('aura-sidebar-visible', sidebarVisible);
  applySidebarVisibility();
}
```

## Task 3: renderSidebar() — tree DOM, expand/collapse, persistence

**Files:**
- Modify: `index.html` (script section)

- [ ] **Step 1: Add `sidebarOpen` state and persistence helpers**

Add this block directly after the `toggleSidebar()` function from Task 2:

```js
// ── Sidebar tree state ────────────────────────────────────────────────────
const sidebarOpen = new Set(
  JSON.parse(localStorage.getItem('aura-sidebar-open') || '[]')
);

function saveSidebarOpen() {
  localStorage.setItem('aura-sidebar-open', JSON.stringify([...sidebarOpen]));
}
```

- [ ] **Step 2: Add `renderSidebar()`**

Add this function after `saveSidebarOpen()`:

```js
function renderSidebar() {
  const container = document.getElementById('tree-content');
  container.innerHTML = '';

  TASKS.forEach(task => {
    const dev      = DEVELOPERS.find(d => d.id === task.developerId);
    const agent    = AGENTS.find(a => a.id === task.agentId);
    const subAgents = agent ? SUB_AGENTS.filter(sa => sa.parentId === agent.id) : [];

    // ── Task row ──
    const taskRow = makeTreeRow(task.id, task.title, '#f59e0b', 0, true);
    container.appendChild(taskRow);

    // ── Dev children ──
    const devChildren = document.createElement('div');
    devChildren.className = 'tree-children';
    devChildren.dataset.parentId = task.id;
    devChildren.style.display = sidebarOpen.has(task.id) ? '' : 'none';

    if (dev) {
      const devRow = makeTreeRow(dev.id, dev.name, 'var(--c-dev)', 1, !!agent);
      devChildren.appendChild(devRow);

      // ── Agent children ──
      const agentChildren = document.createElement('div');
      agentChildren.className = 'tree-children';
      agentChildren.dataset.parentId = dev.id;
      agentChildren.style.display = sidebarOpen.has(dev.id) ? '' : 'none';

      if (agent) {
        const agentRow = makeTreeRow(agent.id, agent.name, 'var(--c-agent)', 2, subAgents.length > 0);
        agentChildren.appendChild(agentRow);

        // ── Sub-agent children ──
        const subChildren = document.createElement('div');
        subChildren.className = 'tree-children';
        subChildren.dataset.parentId = agent.id;
        subChildren.style.display = sidebarOpen.has(agent.id) ? '' : 'none';

        subAgents.forEach(sa => {
          const saRow = makeTreeRow(sa.id, sa.name, 'var(--c-sub)', 3, false);
          subChildren.appendChild(saRow);
        });

        agentChildren.appendChild(subChildren);
      }

      devChildren.appendChild(agentChildren);
    }

    container.appendChild(devChildren);
  });

  updateSidebarActive();
}

function makeTreeRow(id, label, color, depth, hasChildren) {
  const row = document.createElement('div');
  row.className = 'tree-row';
  row.dataset.id = id;
  row.style.paddingLeft = (8 + depth * 12) + 'px';

  const arrow = document.createElement('span');
  arrow.className = 'tree-arrow' + (hasChildren ? '' : ' leaf');
  arrow.textContent = hasChildren ? (sidebarOpen.has(id) ? '▾' : '▸') : '▸';

  const labelEl = document.createElement('span');
  labelEl.className = 'tree-label';
  labelEl.textContent = label;
  labelEl.style.color = color;

  arrow.addEventListener('click', e => {
    e.stopPropagation();
    if (!hasChildren) return;
    if (sidebarOpen.has(id)) {
      sidebarOpen.delete(id);
    } else {
      sidebarOpen.add(id);
    }
    saveSidebarOpen();
    renderSidebar();
  });

  row.addEventListener('click', () => onSidebarRowClick(id));

  row.appendChild(arrow);
  row.appendChild(labelEl);
  return row;
}
```

- [ ] **Step 1: Add `panToNode(id)`**

Add this function after `renderSidebar()`:

```js
function panToNode(id) {
  const pos = positions[id];
  if (!pos) return;
  const canvas = document.getElementById('canvas');
  zoom.x = canvas.clientWidth  / 2 - (pos.x + NODE_W / 2) * zoom.scale;
  zoom.y = canvas.clientHeight / 2 - (pos.y + NODE_H / 2) * zoom.scale;
  applyTransform();
}
```
